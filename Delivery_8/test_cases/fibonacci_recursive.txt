block recursiveFibonacci
receives:
n of type whole
returns whole
{
	if(n == 1 or n == 2){
		return 1;
	}
	else{
		return recursiveFibonacci(n-1) + recursiveFibonacci(n-2);
	}
}

starting block main
{
	variable n of type whole;

	print("Calculate Nth fibonacci number:");
	input(n);

	print(recursiveFibonacci(n));
}
